#include <tunables/global>

profile vilocal flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other files within your app
  /data/** rw,

  # Start new profile for service
  /usr/bin/node cx -> node_runtime,

  profile node_runtime flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_vilocal,

    # Access to options.json and other files within your app
    /data/** rw,

    # Access to mapped volumes specified in config.json
    /share/** rw,

    # Access to app directory
    /app/ r,
    /app/** rmix,

    # Network access for Node.js
    network inet stream,
    network inet dgram,
    network inet6 stream,
    network inet6 dgram,

    # System access
    /usr/bin/node r,
    /usr/bin/yarn rix,
    /usr/bin/jq rix,
    /usr/bin/sed rix,
    /usr/bin/mkfifo rix,
    /usr/bin/tcpdump rix,
    /usr/local/lib/node_modules/ r,
    /usr/local/lib/node_modules/** rm,
    /bin/bash rix,
    /bin/sh rix,
    /bin/mkdir rix,
    /bin/rm rix,
    /bin/cat rix,
    /usr/local/bin/ember-sniff rix,
    /usr/local/bin/whsniff rix,
    /etc/passwd r,
    /etc/group r,
    /etc/nsswitch.conf r,
    /etc/ssl/ r,
    /etc/ssl/** r,
    /root/ r,
    /root/.cache/ rw,
    /root/.cache/node/ rw,
    /root/.cache/node/corepack/ rw,
    /root/.cache/node/corepack/** rw,
    /dev/tty rw,

    # Device access (USB for ZigBee dongles)
    /dev/ttyUSB* rw,
    /dev/ttyACM* rw,
    /dev/bus/usb/ r,
    /dev/bus/usb/** rw,

    # Temporary files
    /tmp/ rw,
    /tmp/** rw,
  }
}
